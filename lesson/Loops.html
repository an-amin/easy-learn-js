JavaScript Loop

  1. for loop
  2. while loop
  3. do-while loop
  4. forEach loop
  5. for in loop

1. For Loop

for( starting_initialise; continue_as_long_as_condition; do_this_each_time )
{
  //some code;
} 
<script>
	for(i=0; i<10; i++)
	{
		console.log(i);
	}
</script>

2. while Loop

var initial_value = value;
while(condition)
{
  //some code;
  //math_expression;
} 
<script>
	var i=0;
	while(i<10)
	{
		console.log(i);
		i++;
	}
</script>

3. do-while Loop

var initial_value = value;
do{
  //some code;
  //math_expression;
}
while(condition) 
<script>
	var i=20;
	do{
		console.log(i);
		i++;
	}
	while(i<10)
</script>

4. forEach Loop

var array = [elements,...];
array.forEach(function(element){
   //some code;
}); 
<script>
	var fruits = ["apple", "banana", "cherry", "grapes"];
	fruits.forEach(function(fruit){
		console.log("I love "+fruit);
	});
</script>
var array = [index=>elements,...];
array.forEach(function(element,index){
   //some code;
}); 
<script>
	var clients = ["Amin", "Hasan", "Farhan"];
	clients.forEach(function(client,serial){
		console.log(client+"'s serial number is "+serial);
	});
</script>

5. for in loop

var anObjectOrArray = {key:element,...};
for( myVariable in anObjectOrArray )
{
  //some code;
}
<script>
	var fruits = {a:"apple", b:"berry", c:"cherry"};
	for(var letter in fruits)
	{
		console.log(letter+' for '+fruits[letter]);
	}
</script>
